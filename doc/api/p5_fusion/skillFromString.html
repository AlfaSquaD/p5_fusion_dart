Skill? skillFromString(String value) {
  switch (value) {
    case 'phys':
      return Skill.phys;
    case 'gun':
      return Skill.gun;
    case 'fire':
      return Skill.fire;
    case 'ice':
      return Skill.ice;
    case 'electric':
      return Skill.electric;
    case 'wind':
      return Skill.wind;
    case 'psy':
      return Skill.psy;
    case 'nuclear':
      return Skill.nuclear;
    case 'bless':
      return Skill.bless;
    case 'curse':
      return Skill.curse;
    case 'almighty':
      return Skill.almighty;
    case 'ailment':
      return Skill.ailment;
    case 'support':
      return Skill.support;
    case 'passive':
      return Skill.passive;
    case 'healing':
      return Skill.healing;
    case 'trait':
      return Skill.trait;
    default:
      return null;
  }
}
